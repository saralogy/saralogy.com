## 2. Content Collection Zod Schemas (`src/content/config.ts`)

```typescript
import { defineCollection, z } from "astro:content";

// ── Shared sub-schemas ────────────────────────────────────────────────────────

const seoFields = z.object({
  title: z.string().max(60),
  description: z.string().min(120).max(160),
  ogImage: z.string().optional(), // path relative to /public, or URL
  noIndex: z.boolean().default(false),
});

const contentStatus = z.enum(["draft", "published"]);

// ── Career Collection ─────────────────────────────────────────────────────────
// Enforces the two-layer architecture. Permanent fields never change after
// publish. Living fields are updated annually (lastReviewed is the signal).

const careerCollection = defineCollection({
  type: "content",
  schema: ({ image }) =>
    z.object({
      // ── Identity ──
      company: z.string(),               // "BSH Hausgeräte GmbH"
      companyShort: z.string(),          // "BSH / Siemens" — for cards
      slug: z.string(),                  // "bsh" — matches filename
      location: z.string(),              // "Munich, Germany"
      period: z.object({
        start: z.string(),               // "2020-05"
        end: z.string().or(z.literal("present")), // "2024-11"
      }),
      logo: image().optional(),          // Company logo (processed by Sharp)

      // ── PERMANENT LAYER (never edit after publish) ──
      role: z.string(),                  // "Senior Brand Communications Manager (Global)"
      roleHistory: z
        .array(
          z.object({
            title: z.string(),
            period: z.string(),
          })
        )
        .optional(),                     // For multi-role tenures like BSH

      wins: z.array(
        z.object({
          headline: z.string(),          // "€1.62M organic media value via purchase-intent content"
          metric: z.string().optional(), // "€1.62M" — for structured display
          context: z.string().optional(), // full sentence with context
        })
      ),                                 // Used to generate WorkExperience schema

      skills: z.array(z.string()),       // ["B2B Marketing", "Brand Strategy", "Agile"]
      tools: z.array(z.string()).optional(), // ["Jira", "Adobe Suite", "TikTok Ads"]
      departureContext: z.string(),      // Honest, professional. Always present.

      // ── LIVING LAYER (updated annually, dated) ──
      currentRelevance: z.object({
        date: z.string(),                // ISO date of last update "2026-06"
        body: z.string(),                // Markdown: "In 2026, this experience is relevant because..."
        audienceAngles: z
          .object({
            recruiter: z.string().optional(),
            customer: z.string().optional(),
            collaborator: z.string().optional(),
          })
          .optional(),
      }),

      projectsFromHere: z
        .array(
          z.object({
            title: z.string(),           // "vision-intelligence"
            url: z.string(),             // "/projects/vision-intelligence"
            connection: z.string(),      // "The gap I saw at BSH became this product."
          })
        )
        .optional(),

      lastReviewed: z.string(),          // ISO date "2026-06-13" — drives the "freshness" signal

      // ── SEO ──
      seo: seoFields,
      status: contentStatus.default("draft"),
      order: z.number().optional(),     // For manual ordering on the index page
    }),
});

// ── Music Collection ──────────────────────────────────────────────────────────
// MusicRecording-ready. Every field maps directly to a schema.org property.

const musicCollection = defineCollection({
  type: "content",
  schema: ({ image }) =>
    z.object({
      title: z.string(),
      slug: z.string(),
      artwork: image().optional(),

      // ── MusicRecording schema fields ──
      byArtist: z.string().default("Berk Saraloglu"),
      duration: z.string().optional(),   // ISO 8601: "PT3M42S"
      genre: z.array(z.string()),        // ["Indie Pop", "Singer-Songwriter"]
      datePublished: z.string().optional(), // ISO date
      isrc: z.string().optional(),       // International Standard Recording Code

      // ── Streaming links ──
      spotify: z.string().url().optional(),
      appleMusic: z.string().url().optional(),
      youtube: z.string().url().optional(),
      soundcloud: z.string().url().optional(),

      // ── Song archaeology (the moat) ──
      releaseStatus: z.enum(["released", "demo", "writing", "archived"]),
      writingProcess: z.string().optional(), // How the song started
      behindTheLyrics: z.string().optional(), // What it's actually about
      productionNotes: z.string().optional(), // Instrumentation, recording approach
      influences: z.array(z.string()).optional(),

      // ── SEO ──
      seo: seoFields,
      status: contentStatus.default("draft"),
    }),
});

// ── Projects Collection ───────────────────────────────────────────────────────
// Non-technical pitch pages. SoftwareApplication schema fields included.

const projectsCollection = defineCollection({
  type: "content",
  schema: ({ image }) =>
    z.object({
      title: z.string(),
      slug: z.string(),
      tagline: z.string().max(100),      // One-sentence pitch
      cover: image().optional(),

      // ── Project status (distinct from contentStatus) ──
      projectStatus: z.enum(["concept", "mvp", "beta", "live", "archived"]),

      // ── Non-technical pitch fields ──
      problem: z.string(),               // The problem it solves (1 sentence)
      audience: z.string(),              // Who it's for
      howItWorks: z.string(),            // Plain-language explanation
      techStack: z.array(z.string()).optional(), // For technical visitors

      // ── Links ──
      githubUrl: z.string().url().optional(),
      liveUrl: z.string().url().optional(),
      waitlistUrl: z.string().url().optional(),

      // ── SoftwareApplication schema fields ──
      applicationCategory: z.string().optional(), // "BusinessApplication"
      operatingSystem: z.string().optional(),      // "Web"
      offers: z
        .object({
          price: z.string(),             // "0" for free, or price
          priceCurrency: z.string().default("EUR"),
        })
        .optional(),

      // ── Career connection (back-links to living layer) ──
      originStory: z.string().optional(), // "This idea started at BSH when..."
      careerOrigin: z.string().optional(), // slug of career page that seeded this

      // ── SEO ──
      seo: seoFields,
      status: contentStatus.default("draft"),
      featured: z.boolean().default(false),
      order: z.number().optional(),
    }),
});

// ── Trips Collection ─────────────────────────────────────────────────────────
// Single collection with a level discriminator — avoids schema duplication.

const tripsCollection = defineCollection({
  type: "content",
  schema: ({ image }) =>
    z.object({
      title: z.string(),                 // "Japan" or "Tokyo"
      slug: z.string(),                  // "japan" or "japan/tokyo"
      level: z.enum(["country", "city"]),
      parentSlug: z.string().optional(), // "japan" for city-level entries
      cover: image().optional(),
      visitedAt: z.string(),             // "2023-11" or "2023-11-15"
      visitCount: z.number().default(1),

      // ── Content ──
      highlight: z.string(),             // 1-sentence hook (used on index)
      recommendation: z.string().optional(), // Genuine rec, not TripAdvisor

      // ── SEO ──
      seo: seoFields,
      status: contentStatus.default("draft"),
    }),
});

// ── Writing Collection (The Intersection essays) ──────────────────────────────
// The `intersects` field enforces the cross-domain format at schema level.

const writingCollection = defineCollection({
  type: "content",
  schema: () =>
    z.object({
      title: z.string(),
      slug: z.string(),
      publishedAt: z.string(),           // ISO date
      updatedAt: z.string().optional(),

      // ── Intersection format ──
      intersects: z
        .array(
          z.enum([
            "marketing",
            "music",
            "ai",
            "product",
            "career",
            "design",
            "culture",
          ])
        )
        .min(2),                         // Enforces cross-domain at schema level

      // ── Content metadata ──
      excerpt: z.string().max(280),      // Twitter-card length summary
      readingTime: z.number().optional(), // minutes

      // ── SEO ──
      seo: seoFields,
      status: contentStatus.default("draft"),
      featured: z.boolean().default(false),
    }),
});

// ── Export ───────────────────────────────────────────────────────────────────

export const collections = {
  career: careerCollection,
  music: musicCollection,
  projects: projectsCollection,
  trips: tripsCollection,
  writing: writingCollection,
};
```

## 5. Subdomain Plan

| Subdomain | Status | Action Now | Purpose |
|---|---|---|---|
| `saralogy.com` | Active | Configure DNS → Vercel | Main site |
| `www.saralogy.com` | Active (redirect) | Vercel redirects www → apex | SEO canonical |
| `app.saralogy.com` | Parked | CNAME → Vercel (no project) | AI service product (vision-intelligence) — activate when ready |
| `api.saralogy.com` | Parked | CNAME → Vercel (no project) | API for AI service — activate when ready |
| `blog.saralogy.com` | **Deferred** | Do NOT configure yet | Would split SEO authority from main domain. Writing lives at `/writing`, not on a subdomain. |

**Why blog subdomain is explicitly deferred:** Every page on a subdomain builds authority on the subdomain, not on `saralogy.com`. For a site in early authority-building phase, splitting authority is a net negative. The `/writing` section on the main domain compounds authority. Revisit at 24 months if the site needs dedicated CMS infrastructure.

## 6. Architecture Decisions Log

| # | Decision | Rationale | Alternative Rejected |
|---|---|---|---|
| 1 | **Content collections over CMS** | ≤3h/week content cadence; no team; Astro collections are type-safe and git-versioned | Contentful / Sanity: adds API latency, cost, external dependency for a one-person operation |
| 2 | **MDX over MD** | Allows embedding React/Astro components inside content (song player, interactive maps) without moving to a different system | Plain MD: can't embed interactive components. Full Astro: requires a page file per content entry (no bulk authoring) |
| 3 | **Single `trips` collection with `level` discriminator** | Avoids schema duplication for country vs city entries; `parentSlug` enables tree traversal | Two collections (`trips-country`, `trips-city`): redundant schemas, more config, harder to query together |
| 4 | **`intersects: z.array(...).min(2)` on writing** | Schema-level enforcement of "The Intersection" format — makes it impossible to publish an essay without naming two domains | No enforcement: format degrades over time without the constraint |
| 5 | **`wins` as structured array in career** | Enables programmatic generation of WorkExperience schema from structured data, not raw prose | Free text only: loses the machine-readable structure needed for AEO |
| 6 | **Two-layer architecture as schema contract** | `currentRelevance` and `lastReviewed` are required fields, not optional. Forces the living layer to be present at publish. | Optional fields: authors skip the living layer, pages decay |
| 7 | **`seoFields` sub-schema on every collection** | Centralizes SEO validation; enforces title ≤60 chars and description 120–160 chars at content authoring time | Ad-hoc meta on each collection: drifts, inconsistent limits, validation only at render time |
| 8 | **`/lib/schema/` as separate module** | JSON-LD builders are TypeScript functions, not inline strings. Type-safe, testable, reusable across page types | Inline JSON-LD in each layout: duplicated, drifts, can't be unit tested |
| 9 | **`src/pages/ai-marketing-dach.astro` as static page** | Pillar page needs hand-crafted structure, schema markup, and internal links — not suitable for a generic collection template | Putting it in a `writing` collection: loses flexibility, forces it into the essay template |
| 10 | **Path aliases in tsconfig** | `@components/`, `@layouts/`, `@lib/` import aliases prevent `../../../../` chains | Relative imports: fragile, hard to refactor when moving files |
| 11 | **`artwork` and `cover` use Astro `image()` schema** | Enables Sharp-based optimization at build time; generates responsive srcsets automatically | Storing as URLs: bypasses optimization, increases page weight |
| 12 | **`status: draft` as default** | Nothing publishes accidentally; explicit intent required to go live | `status: published` as default: risk of partial drafts appearing in production |
| 13 | **`src/lib/schema/` not `src/components/seo/`** | JSON-LD builders are pure TypeScript functions (data → string), not UI components. Separating concerns: data in lib, rendering in components | Putting JSON-LD builders in components: mixes data construction with rendering logic |
| 14 | **`blog.saralogy.com` subdomain deferred** | SEO authority concentration on main domain is the priority for the first 24 months. Subdomain = authority split | Subdomain from day one: wastes early link equity by splitting it across domains |

## 8. Implementation Sequence (10 steps, each independently deployable)

1. **Scaffold** — `bunx create astro@latest saralogy` (use the Empty template), install Tailwind + sitemap, commit `astro.config.mjs`, `vercel.json`, `tsconfig.json`, `.github/workflows/ci.yml`
2. **Base layout** — `Base.astro`, `global.css`, `Header.astro`, `Footer.astro`; deploy to Vercel and verify CI passes
3. **Content collections** — `src/content/config.ts` with all 5 schemas; create stub MDX files for each collection type
4. **Homepage** — `index.astro` with intent router and basic sections; confirm routing works
5. **About page** — `about.astro` with Person schema, FAQ block; this is MVP page #1
6. **Career section** — `[slug].astro` template + `bsh.mdx` content; two-layer architecture live; this is MVP page #2
7. **Pillar page** — `ai-marketing-dach.astro` with full content and internal links; this is MVP page #3
8. **Music section** — `music/index.astro` + `[slug].astro` + first 2 song pages
9. **Projects section** — `projects/index.astro` + `[slug].astro` + vision-intelligence + resume-matcher
10. **Launch prep** — `/now`, `/work`, sitemap submission to Search Console, analytics
